
idade = int(input(print('Digite a altura: ')))

if idade < 5:
    print("Menor de 5 anos")

if (idade >= 5) and (idade <= 7):
    print("Infantil")

if (idade >= 7) and (idade <=10):
    print("Juvenil")

if (idade >= 11) and (idade <=15):
    print("Adolescente")

if (idade >= 16) and (idade <=30):
    print( "Adulto")

if idade > 30:
    print("Senior")
